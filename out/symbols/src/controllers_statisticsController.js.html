 11 // Endpoint que cuenta los documentos donde el conductor (rider) tiene el correo del usuario
 12 // Utiliza la funcion countDocuments de Mongoose
 13 // Caso borde: El carnet no esta asociado a ningun conductor -> La consulta devuelve 0
 14 async function getRidesGiven (req, res) {
 15   const validate = validateIn(req.secret, emailRules, emailMessage)
 16   if (!validate.pass) {
 17     const message = 'Los datos no cumplen con el formato requerido'
 18     logger.log('error', message, {user: req.secret.email, operation: 'query-rides-given', status: 401})
 19     return res.status(401).send(response(false, validate.errors, message))
 20   }
 21   
 22   const { status, data, message } = await rides.countDocuments({ 
 23     rider: req.secret.email 
 24   }).then(callbackCount)
 25   logger.log('info', 'Consulta nro de colas dadas', {user: req.secret.email, operation: 'query-rides-given', status: 200})
 26   return res.status(status).send(response(status === 200, data, message))
 27 }

 47 // Endpoint que cuenta los documentos donde el usuario ha sido conductor de la cola y suma el numero total de likes
 48 // Utiliza la funcion aggregate de Mongoose donde $match recibe el filtro de la consulta
 49 // $unwind Convierte el arreglo seleccionado (comments) en documentos unicos para ser contados
 50 // $group Agrupa los documentos por id (o revisa todos si se coloca null)
 51 // $sum acummulador
 52 // Caso borde: El usuario no aparece como conductor -> La consulta devuelve 0
 53 // Caso borde: El conductor no ha recibido likes -> La consulta devuelve 0
 54 async function getLikesCount (req, res) {
 55   const validate = validateIn(req.secret, emailRules, emailMessage)
 56   if (!validate.pass) {
 57     const message = 'Los datos no cumplen con el formato requerido'
 58     logger.log('error', message, {user: req.secret.email, operation: 'query-likes', status: 401})
 59     return res.status(401).send(response(false, validate.errors, message))
 60   }
 61   
 62   const { status, data, message } = await rides.aggregate([
 63     { $match: { rider: req.secret.email, ride_finished: true } },
 64     { $unwind: '$comments' },
 65     { $match: {'comments.like': true } },
 66     { $group: { _id: null, count: { $sum: 1 } } },
 67     { $project: {_id: 0} }
 68   ]).then(callbackAggregation)
 69   logger.log('info', 'Consulta nro de likes', {user: req.secret.email, operation: 'query-likes', status: 200})
 70   return res.status(status).send(response(status === 200, data, message))
 71 }
