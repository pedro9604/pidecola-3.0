134 /**
135  * Endpoint que crea una solicitud de cola.
136  * No debería modificarse a no ser que se cambie toda lógica detrás del
137  * algoritmo de recomendación.
138  * @author Francisco Márquez <12-11163@usb.ve>

140  * @async
141  * @param {Object} req - Un HTTP Request
142  * @param {Object} res - Un HTTP Response
143  * @returns {Object}

145 async function create (req, res) {
146   const { status, errors, message } = verifyRequest(req.body)
147   if (!status) {
148     logger.log('error', message, {user: req.body.user, operation: 'create-request', status: 400})
149     return res.status(400).send(response(false, errors, message))
150   }
151   const user = await users.findByEmail(req.body.user).then(callback)
152   const request = {
153     email: req.body.user,
154     user: {
155       usbid: user.email.split('@')[0],
156       phone: user.phone_number,
157       fName: user.first_name,
158       lName: user.last_name,
159       major: user.major,
160       prPic: user.profile_pic
161     },
162     startLocation: req.body.startLocation,
163     destination: req.body.destination,
164     comment: req.body.comment,
165     im_going: req.body.im_going,
166     status: true
167   }
168   const insert = add(request)
169   logger.log('info', 'Solicitud creada', {user: req.body.user, operation: 'create-request', status: 200})
170   return res.status(200).send(response(true, insert, 'Solicitud exitosa'))
171 }

173 /**
174  * Función que verifica la validez de los datos de una solicitud para
175  * modificar la lista de solicitudes de cola.
176  * @author Francisco Márquez <12-11163@usb.ve>
177  * @private
178  * @param {string} request
179  * @returns {Verification}
180  */
181 function verifyRequest (request, cancel = false) {
182   const validate = validateIn(request, requestsRules, errorsMessage)
183   const fromUSB = request.startLocation === 'USB'
184   const toUSB = request.destination === 'USB'
185   const start = fromNameToInt(request.startLocation) > -2
186   const dest = fromNameToInt(request.destination) > -2
187   const exists = alreadyRequested(request.user).in
188   var errors = '', message = ''
189   if (!cancel) {
190     if (!(validate.pass && !exists && (fromUSB !== toUSB) && start && dest)) {
191       if (!validate.pass) {
192         errors = validate.errors
193         message = 'Los datos introducidos no cumplen con el formato requerido'
194       } else if (exists) {
195         errors = 'Solicitud existente'
196         message = 'No puedes solicitar más de una cola'
197       } else if (fromUSB) {
198         errors = 'Cola es desde la USB'
199         message = 'No puede haber una cola desde la USB hasta la USB'
200       } else if (!toUSB) {
201         errors = 'Cola no involucra a la USB'
202         message = 'No puede haber una cola que no salga de, o llegue a, la USB'
203       } else if (!start) {
204         errors = 'Cola no empieza en una parada autorizada'
205         message = 'Ninguna cola puede involucrar una parada no autorizada'
206       } else {
207         errors = 'Cola no termina en una parada autorizada'
208         message = 'Ninguna cola puede involucrar una parada no autorizada'
209       }
210     } else {
211       return { status: true, errors: '', message: '' }
212     }
213   } else {
214     if (!(validate.pass && exists && (fromUSB !== toUSB) && start && dest)) {
215       if (!validate.pass) {
216         errors = validate.errors
217         message = 'Los datos introducidos no cumplen con el formato requerido'
218       } else if (!exists) {
219         errors = 'Solicitud inexistente'
220         message = 'No tienes una cola que cancelar'
221       } else if (fromUSB) {
222         errors = 'Cola es desde la USB'
223         message = 'No puede haber una cola desde la USB hasta la USB'
224       } else if (!toUSB) {
225         errors = 'Cola no involucra a la USB'
226         message = 'No puede haber una cola que no salga de, o llegue a, la USB'
227       } else if (!start) {
228         errors = 'Cola no empieza en una parada autorizada'
229         message = 'Ninguna cola puede involucrar una parada no autorizada'
230       } else {
231         errors = 'Cola no termina en una parada autorizada'
232         message = 'Ninguna cola puede involucrar una parada no autorizada'
233       }
234     } else {
235       return { status: true, errors: '', message: '' }
236     }
237   }
238   return { status: false, errors: errors, message: message }
239 }

241 /**
242  * Indica si el email se encuentra en la lista requestsList.
243  * No debería modificarse a no ser que se cambie toda lógica detrás del
244  * algoritmo de recomendación.
245  * @author Francisco Márquez <12-11163@usb.ve>
246  * @private
247  * @param {string} email - Un email
248  * @returns {Elem}
249  */
250 function alreadyRequested (email) {
251   for (var i = 0; i < requestsList.length; i++) {
252     if (inList(email, requestsList[i].requests).in) {
253       return inList(email, requestsList[i].requests)
254     }
255   }
256   return { in: false, elem: {} }
257 }

276 /**
277  * Función que agrega una solicitud de cola a su respectiva parada.
278  * @author Francisco Márquez <12-11163@usb.ve>
279  * @private
280  * @param {Object} newRequest
281  * @returns {Object} Solicitud insertada o solicitud recibida si hay error
282  */
283 function add (newRequest) {
284   const fromUSB = newRequest.startLocation === 'USB'
285   let index
286   if (fromUSB) {
287     index = fromNameToInt(newRequest.destination)
288   } else {
289     index = fromNameToInt(newRequest.startLocation)
290   }
291   requestsList[index].requests.push(newRequest)
292   client.sadd(requestsList[index].name, JSON.stringify(newRequest))
293   handleSockets.sendPassengers(requestsList[index].name)
294   return newRequest
295 }

301 /**
302  * Endpoint para cancelar una solicitud de cola.
303  * No debería modificarse a no ser que se cambie toda lógica detrás del
304  * algoritmo de recomendación.
305  * @author Francisco Márquez <12-11163@usb.ve>

307  * @param {Object} req - Un HTTP Request
308  * @param {Object} res - Un HTTP Response
309  * @returns {Object}
310  */
311 function cancel (req, res) {
312   const { status, errors, message } = verifyRequest(req.body, true)
313   if (!status) {
314     logger.log('error', message, {user: req.body.email, operation: 'delete-request', status: 400})
315     return res.status(400).send(response(false, errors, message))
316   }
317   var del = remove(req.body, true)
318   if (del) {
319     logger.log('info', 'Solicitud eliminada', {user: req.body.email, operation: 'delete-request', status: 200})
320     return res.status(200).send(response(true, '', 'Solicitud exitosa'))
321   } else {
322     logger.log('error', 'La solicitud no existe', {user: req.body.email, operation: 'delete-request', status: 500})
323     return res.status(200).send(response(false, '', 'Cola no existe'))
324   }
325 }

327 /**
328  * Función que elimina una solicitud de cola a su respectiva parada.
329  * No debería modificarse a no ser que se cambie toda lógica detrás del
330  * algoritmo de recomendación.
331  * @author Francisco Márquez <12-11163@usb.ve>
332  * @private
333  * @param {Object} deleteRequest
334  * @returns {boolean} true si y solo si fue correctamente eliminada
335  */
336 function remove (deleteRequest, removeList = false) {
337   let index
338   const fromUSB = deleteRequest.startLocation === 'USB'
339   if (fromUSB) {
340     index = fromNameToInt(deleteRequest.destination)
341   } else {
342     index = fromNameToInt(deleteRequest.startLocation)
343   }
344   if(!requestsList[index].requests) return false
345   for (var i = 0; i < requestsList[index].requests.length; i++) {
346     const req = requestsList[index].requests[i]
347     const email = req.email
348     if (email === deleteRequest.user || email === deleteRequest.email) {
349       requestsList[index].requests.splice(i, 1)
350       client.srem(requestsList[index].name, JSON.stringify(req))
351       if (removeList) {
352         handleSockets.sendPassengers(requestsList[index].name)
353       }
354       return true
355     }
356   }
357   return false
358 }

597  * Endpoint para responder una oferta de cola.

605  * @returns {Object}
606  */
607 async function respondOfferRide (req, res) {
608   const { status, errors, message } = verifyRespondOffer(req.body)
609   if (!status) {
610     logger.log('error', message, {user: req.body.rider, operation: 'respond-request', status: 400})

612   }
613   const answer = await respondOffer(req.body)

616     logger.log('info', 'Oferta respondida', {user: req.body.rider, operation: 'respond-request', status: 200})
617     return res.status(200).send(response(true, answer.log, 'Respondiste'))
618   } else {
619     logger.log('error', 'Error en respuesta', {user: req.body.rider, operation: 'respond-request', status: 500})
620     return res.status(500).send(response(false, answer.errors, 'Error'))
621   }
622 }

624 /**
625  * Función que verifica que los datos recibidos tengan el formato adecuado para
626  * No debería modificarse a no ser que se cambie toda lógica detrás del
627  * algoritmo de recomendación.
628  * @author Francisco Márquez <12-11163@usb.ve>
629  * @private
630  * @param {Object} dataResponse
631  * @param {string} dataResponse.rider     - El correo del ofertante
632  * @param {string} dataResponse.passenger - El correo del solicitante
633  * @param {string} dataResponse.accept    - String = 'Sí' || 'No'
634  * @returns {Verification}
635  */
636 function verifyRespondOffer (dataResponse) {
637   const responseRules = {
638     rider: 'required|email',
639     passenger: 'required|email',
640     accept: 'required|string'
641   }
642   const offerMessage = {
643     'required.rider': 'El conductor es necesario',
644     'required.passenger': 'El pasajero es necesario',
645     'required.accept': 'La respuesta del solicitante es necesaria'
646   }
647   let errors
648   let message
649   const validate = validateIn(dataResponse, responseRules, offerMessage)
650   const yesOrNot = dataResponse.accept === 'Sí' || dataResponse.accept === 'No'
651   if (!(validate.pass && yesOrNot)) {
652     if (!validate) {
653       errors = validate.errors
654       message = 'Los datos introducidos no cumplen con el formato requerido'
655     } else {
656       errors = 'La respuesta no parece ser Sí o No'
657       message = 'La respuesta tiene que ser Sí o No'
658     }
659     return { status: false, errors: errors, message: message }
660   }
661   return { status: true, errors: '', message: '' }
662 }

679   const name = await users.findByEmail(response.rider).then(callback)
680   const html = responseTemplate(name.first_name)
681   if (response.accept === 'Sí') {

684         if (err) return console.log(err)

686         list.forEach( offer => {

688         })
689       })
690       try {

695     } else {
696       const log = 'Parece que la solicitud de cola no existe'
697       const errors = 'La solicitud de cola no fue encontrada'
698       return { sent: false, log: log, errors: errors }
699     }
700   } else {
701     client.del('OFFER-' + response.passenger + '-' + response.rider)
702     try {
703       return sendEmail(response.rider, subj, html).then(callbackMail) 
704     } catch (error) {
705       console.log('ERROR SENDING EMAIL');
