 43  * Endpoint para crear una nueva cola.
 44  * No debería modificarse a no ser que se cambie toda lógica detrás del
 45  * algoritmo de recomendación.
 46  * @author Francisco Márquez <12-11163@usb.ve>
 47  * @public
 48  * @async
 49  * @param {Object} req - Un HTTP Request
 50  * @param {Object} res - Un HTTP Response
 51  * @returns {Object}
 52  */
 53 async function create (req, res) {
 54   const { status, errors, message } = await verifyDataRide(req.body)
 55   if (!status) {
 56     logger.log('error', message, {user: req.body.rider, operation: 'create-ride', status: 400})
 57     return res.status(400).send(response(false, errors, message))
 58   }
 59   const rideInf = await newRide(req.body)
 60   if (rideInf) {
 61     sendPassengers(req.body.passenger)
 62     logger.log('info', 'Cola creada', {user: req.body.rider, operation: 'create-ride', status: 200})
 63     return res.status(200).send(response(true, rideInf, 'Cola creada'))
 64   } else {
 65     logger.log('error', 'Error en creacion de cola', {user: req.body.rider, operation: 'create-ride', status: 500})
 66     return res.status(500).send(response(true, rideInf, 'Cola no fue creada'))
 67   }
 68 }

 70 async function cancel (req, res) {
 71   const {rider, passengers} = req.body
 72   passengers.forEach( passenger => {
 73     handleSockets.cancelOffer(rider, passenger)
 74   })
 75   return res.status(200).send(response(true, 'Ok', 'Cola cancelada.'))

 87 /**
 88  * Función que verifica la validez de los datos de una solicitud para
 89  * modificar la base de datos de las colas.
 90  * @author Francisco Márquez <12-11163@usb.ve>
 91  * @private
 92  * @async
 93  * @param {string} status
 94  * @returns {Verification}
 95  */
 96 async function verifyDataRide (dataRide) {
 97   const validate = validateIn(dataRide, rideRules, errorsMessage)
 98   const arrayPas = Array.isArray(dataRide.passenger)
 99   var emptyPas = true
100   var pass = true
101   var validPass = true
102   const valSeats = parseInt(dataRide.seats) >= dataRide.passenger.length
103   const rider = await findByEmail(dataRide.rider).then(callback) != null
104   if (dataRide.passenger.length > 0) {
105     pass = !(dataRide.passenger.find(user => user.email === dataRide.rider))
106     for (var i = 0; i < dataRide.passenger.length; i++) {
107       if (await findByEmail(dataRide.passenger[i].email) === null) {
108         validPass = false
109       }
110     }
111   } else {
112     emptyPas = false
113   }
114   if (!(validate.pass && arrayPas && emptyPas && pass && validPass && valSeats && rider)) {
115     var errors = ''
116     var message = ''
117     if (!validate.pass) {
118       errors = validate.errors
119       message = 'Los datos introducidos no cumplen con el formato requerido'
120     } else if (!arrayPas) {
121       errors = 'Los pasajeros no están en un arreglo'
122       message = 'Los pasajeros deben estar en un arreglo'
123     } else if (!emptyPas) {
124       errors = 'Arreglo vacío de pasajeros'
125       message = 'No puede haber una cola sin pasajeros'
126     } else if (!pass) {
127       errors = 'El conductor es un pasajero'
128       message = 'El conductor no puede ser un pasajero'
129     } else if (!validPass) {
130       errors = 'Hay pasajeros no registrados en esta cola'
131       message = 'Todo pasajero tiene que ser un usuario registrado'
132     } else if (!valSeats) {
133       errors = 'Hay más pasajeros que asientos disponibles'
134       message = 'Debe haber a lo sumo tantos pasajeros como asientos haya'
135     } else {
136       errors = 'El conductor no está registrado'
137       message = 'El conductor tiene que estar registrado'
138     }
139     return { status: false, errors: errors, message: message }
140   }
141   return { status: true, errors: '', message: '' }
142 }

152 async function newRide (dataRide) {
153   const ride = {
154     rider: dataRide.rider,
155     passenger: dataRide.passenger,
156     vehicle: dataRide.vehicleId,
157     available_seats: dataRide.seats,
158     status: 'En Espera',

160     destination: dataRide.destination,
161     time: new Date(),
162     ride_finished: false,
163     comments: []
164   }
165   return rides.create(ride).then(callback)

168 function sendPassengers (passengers) {
169   for (var i = 0; i < passengers.length; i++) {
170     const index = cast(passengers[i].ruta)
171     for (var j = 0; j < list[index].requests.length; j++) {
172       if (list[index].requests[j].email === passengers[i].email) {
173         handleSockets.sendPassengers(list[index].name)
174         break
175       }
176     }
177   }
178 }

195 async function endRide (req, res) {
196   const { status, errors, message } = await verifyDataRide(req.body)
197   if (!status) {
198     logger.log('error', message, {user: req.body.rider, operation: 'end-ride', status: 400})
199     return res.status(400).send(response(false, errors, message))
200   }
201   const query = { $set: { ride_finished: true, status: 'Finalizado' } }
202   const rideInf = await updateRide(req.body, query)
203   if (rideInf) {
204     logger.log('info', 'Cola finalizada', {user: req.body.rider, operation: 'end-ride', status: 200})
205     return res.status(200).send(response(true, rideInf, 'Cola finalizada'))
206   } else {
207     logger.log('error', 'Error en finalizar cola', {user: req.body.rider, operation: 'end-ride', status: 500})
208     return res.status(500).send(response(false, '', 'Cola no existe'))
209   }
210 }

269 /**
270  * Función que verifica la validez del estado de una cola para cambiarlo.
271  * @author Francisco Márquez <12-11163@usb.ve>
272  * @private
273  * @param {string} status
274  * @returns {Verification}
275  */
276 async function verifyStatusRide (statusRide) {
277   const { status, errors, message } = await verifyDataRide(statusRide)
278   if (!status) return { status: status, errors: errors, message: message }
279   const rule = { status: 'required|string' }
280   const mssg = { 'required.status': 'El estado de la solicitud es requerido' }
281   const validate = validateIn(statusRide, rule, mssg)
282   if (!validate.pass) {
283     return {
284       status: false,
285       errors: validate.errors,
286       message: 'El estado no tiene el formato válido'
287     }
288   } else if (statusRide.status === 'En Espera') {
289     return { status: true, errors: '', message: '' }
290   } else if (statusRide.status === 'En Camino') {
291     return { status: true, errors: '', message: '' }
292   } else if (statusRide.status === 'Accidentado') {
293     return { status: true, err: '', message: '' }
294   } else if (statusRide.status === 'Finalizado') {
295     return { status: true, errors: '', message: '' }
296   } else {
297     return {
298       status: false,
299       errors: validate.errors,
300       message: 'El estado no tiene el formato válido'
301     }
302   }
303 }

338 /**
339  * Función que verifica la validez de los datos de una solicitud para
340  * agregar comentarios a una cola ya finalizada.
341  * @author Francisco Márquez <12-11163@usb.ve>
342  * @private
343  * @async
344  * @param {string} dataRide
345  * @returns {Verification}
346  */
347 async function verifyComments (dataRide) {
348   const rules = {
349     rider: 'required|email',
350     user: 'required|email',
351     startLocation: 'required|string',
352     destination: 'required|string',
353     like: 'required|string',
354     comment: 'string'
355   }
356   const mssg = {
357     'required.rider': 'El correo del conductor es necesario',
358     'required.user': 'El correo del pasajero es necesario',
359     'required.startLocation': 'El lugar de partida es necesario',
360     'required.destination': 'El lugar de destino es necesario',
361     'required.like': 'El like es necesario'
362   }
363   const validate = validateIn(dataRide, rules, mssg)
364   const like = dataRide.like === 'Sí' || dataRide.like === 'No'
365   const rider = await findByEmail(dataRide.rider).then((sucs, err) => {
366     return !err && !!sucs
367   })
368   if (!(validate.pass && like && rider)) {
369     var errors = ''
370     var message = ''
371     if (!validate.pass) {
372       errors = validate.errors
373       message = 'Los datos introducidos no cumplen con el formato requerido'
374     } else if (!like) {
375       errors = 'Parece que el like no es Sí ni No'
376       message = 'El like debe ser Sí o No'
377     } else {
378       errors = 'Parece que el conductor no se encuentra registrado'
379       message = 'El conductor debe ser un usuario registrado'
380     }
381     return { status: false, errors: errors, message: message }
382   }
383   return { status: true, errors: '', message: '' }
384 }

386 /**
387  * Función que inserta un comentario en una cola ya finalizada.
388  * @author Francisco Márquez <12-11163@usb.ve>
389  * @private
390  * @async
391  * @param {string} dataRide
392  * @returns {Object}
393  */
394 async function comment (dataRide) {
395   const user = dataRide.user
396   const like = dataRide.like === 'Sí'
397   const comn = !like ? dataRide.comment : undefined
398   const data = {
399     rider: dataRide.rider,
400     start_location: dataRide.startLocation,
401     destination: dataRide.destination,
402     status: 'Finalizado',
403     ride_finished: true
404   }
405   const query = { user_email: user, like: like, dislike: !like, comment: comn }
406   return rides.findOne(data, {}, { sort: { '_id': -1 } }).then((sucs, err) => {
407     if (!err && sucs) {
408       for (var i = 0; i < sucs.comments.length; i++) {
409         if (sucs.comments[i].user_email === query.user_email) {
410           return sucs
411         }
412       }
413       sucs.comments.push(query)
414       sucs.markModified('comments')
415       sucs.save().then(sucs => sucs).catch(error => console.log(error))
416       return sucs
417     } else {
418       return err
419     }
420   })
421 }

427 /**
428  * Endpoint para obtener información de una cola.
429  * No debería modificarse a no ser que se cambie toda lógica detrás del
430  * algoritmo de recomendación.
431  * @author Francisco Márquez <12-11163@usb.ve>
432  * @public
433  * @async
434  * @param {Object} req - Un HTTP Request
435  * @param {Object} res - Un HTTP Response
436  * @returns {Object}
437  */
438 async function getRide (req, res) {
439   const { status, errors, message } = verifyGetRide(req.secret)
440   if (!status) {
441     logger.log('error', message, {user: req.secret.email, operation: 'get-ride', status: 400})
442     return res.status(400).send(response(false, errors, message))
443   }
444   var rideInf = await findRide(req.secret.email)
445   if (rideInf) {
446     const rider = await findByEmail(rideInf.rider).then(callback)
447     const riderInfo = {
448       phone: rider.phone_number,
449       fname: rider.first_name,
450       lname: rider.last_name,
451       age: rider.age,
452       major: rider.major,
453       photo: rider.profile_pic,
454       vehicle: await users.aggregate([
455         { $unwind: '$vehicles' },
456         { $match: {'vehicles._id': rideInf.vehicle }},
457         { $project: {_id: 0, vehicles: 1}} 
458       ]).then(callback)
459     }
460     data = { ride: rideInf, riderInfo: riderInfo }
461     logger.log('info', 'Cola encontrada', {user: req.secret.email, operation: 'get-ride', status: 200})
462     return res.status(200).send(response(true, data, ''))
463   }
464   logger.log('info', 'Cola no existe', {user: req.secret.email, operation: 'get-ride', status: 206})
465   return res.status(206).send(response(true, 'Cola no existe', 'La cola buscada no está registrada'))
466 }
467 
468 function verifyGetRide (request) {
469   const validate = validateIn(request, emailRules, emailMessage)
470   var errors = '', message = ''
471   if (!validate.pass) {
472     errors = validate.errors
473     message = 'Los datos introducidos no cumplen con el formato requerido'
474   }
475   return { status: errors === '', errors: errors, message: message }
476 }

478 /**
479  * Función que devuelve una cola de la base de datos.
480  * @author Francisco Márquez <12-11163@usb.ve>
481  * @private
482  * @async
483  * @param {Object} dataRide - Datos de la cola a modificar
484  * @returns {Object} Datos de la cola insertada en la base de datos
485  */
486 async function findRide (email) {
487   const psgr = { $elemMatch: { email: email } }
488   const ride = await rides.findOne({ rider: email, ride_finished: false })
489   const pass = await rides.findOne({ passenger: psgr, ride_finished: false })
490   return ride || pass
491 }
