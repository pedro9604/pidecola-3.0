 74 /**
 75  * Indica si la parada elem está en el conjunto set.
 76  * No debería modificarse a no ser que se cambie toda lógica detrás del
 77  * algoritmo de recomendación.
 78  * @author Francisco Márquez <12-11163@usb.ve>
 79  * @private
 80  * @param {string} elem - Una parada
 81  * @param {Object} set  - Un conjunto de dos paradas
 82  * @returns {boolean}
 83  */
 84 function member (elem, set) {
 85   return set.from === elem || set.to === elem
 86 }
 87 
 88 /**
 89  * Indica si la parada elem no está en el conjunto set.
 90  * No debería modificarse a no ser que se cambie toda lógica detrás del
 91  * algoritmo de recomendación.
 92  * @author Francisco Márquez <12-11163@usb.ve>
 93  * @param {Object} set  - Un conjunto de dos paradas
 94  * @returns {boolean}
 95  */
 96 function notMember (elem, set) {
 97   return !member(elem, set)
 98 }

100 /**
101  * Retorna todos los elementos en set que cumplen con predicate.
102  * No debería modificarse a no ser que se cambie toda lógica detrás del
103  * algoritmo de recomendación.
104  * @author Francisco Márquez <12-11163@usb.ve>
105  * @private
106  * @param {Array}    set       - Un arreglo
107  * @param {function} predicate - Un función que devuelve un booleano
108  * @returns {Array}
109  */
110 function filter (set, predicate) {
111   const filtered = []
112   for (let i = 0; i < set.length; i++) {
113     if (predicate(set[i])) {
114       filtered.push(set[i])
115     }
116   }
117   return filtered
118 }

120 /**
121  * Retorna las paradas en a que no son b
122  * No debería modificarse a no ser que se cambie toda lógica detrás del
123  * algoritmo de recomendación.
124  * @author Francisco Márquez <12-11163@usb.ve>
125  * @private
126  * @param {Object} a - Un conjunto de dos paradas
127  * @param {string} b - Una parada
128  * @returns {string|Object}
129  */
130 function difference (a, b) {
131   if (a.from === b) {
132     return a.to
133   } else if (a.to === b) {
134     return a.from
135   } else {
136     return a
137   }
138 }

140 /**
141  * Implementación del algoritmo de recomendación.
142  * El primer elemento es la parada recibida y la cola está ordenada
143  * crecientemente en distancia respecto de la parada recibida
144  * No debería modificarse a no ser que se cambie toda lógica detrás del
145  * algoritmo de recomendación.
146  * @author Francisco Márquez <12-11163@usb.ve>
147  * @private
148  * @param {string} stop - Una parada
149  * @returns {Stop[]}
150  */
151 function prioridad (stop) {
152   const output = [requests.requestsList[requests.cast(stop)]]
153   const f = filter(stops, e => {
154     return member(stop, e.stops) && notMember('USB', e.stops)
155   })
156   const m = f.map(e => {
157     return { detour: difference(e.stops, stop), distance: e.distance }
158   })
159   const s = m.sort((dest1, dest2) => {
160     if (dest1.distance < dest2.distance) {
161       return -1
162     } else if (dest1.distance > dest2.distance) {
163       return 1
164     } else {
165       return 0
166     }
167   })
168   for (let i = 0; i < s.length; i++) {
169     output.push(requests.requestsList[requests.cast(s[i].detour)])
170   }
171   return output
172 }

174 /**
175  * Endpoint para recomendar solicitudes de cola.
176  * No debería modificarse a no ser que se cambie toda lógica detrás del
177  * algoritmo de recomendación.
178  * @author Francisco Márquez <12-11163@usb.ve>
179  * @public
180  * @param {Object} req - Un HTTP Request
181  * @param {Object} res - Un HTTP Response
182  * @returns {Object}
183  */
184 function recommend (req, res) {
185   try {
186     const info = prioridad(req.body.destination)
187     logger.log('info', 'Recomendacion de pasajeros realizada', {user: 'System', operation: 'recommend', status: 200})
188     return res.status(200).send(response(true, info, ''))
189   } catch (error) {
190     logger.log('error', `Error en recomendacion: ${error}`, {user: 'System', operation: 'recommend', status: 400})
191     return res.status(400).send(response(false, {}, 'Ha ocurrido un error'))
192   }
193 }

195 function getPriority (destination) {
196   return prioridad(destination)
197 }
