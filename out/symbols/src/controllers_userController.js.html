 52 /**
 53  * Endpoint para crear un nuevo usuario. El número de veces que son calculados
 54  * los BCrypt hash es 12
 55  * No debería modificarse a no ser que se cambie toda lógica detrás del
 56  * algoritmo de recomendación.
 57  * @async
 58  * @public
 59  * @param {Object} req - Un HTTP Request
 60  * @param {Object} res - Un HTTP Response
 61  * @returns {Object}
 62  */
 63 async function create (req, res) {
 64   const { code, status, errors, message } = await verifyCreate(req.body)
 65   if (!status) {
 66     logger.log('error', message, {user: req.body.email, operation: 'create', status: code})
 67     return res.status(code).send(response(false, errors, message))
 68   }
 69   req.body.password = await bcrypt.hash(req.body.password, 12)
 70   const user = await addUser(req.body).then((sucs, err) => {
 71     if (!err && sucs) return { code: 200, user: sucs }
 72     return { code: err.code === 11000 ? 11000 : 500, user: null }
 73   })
 74   let mssg
 75   if (user.code === 200) {
 76     const create = await responseCreate(user.user)
 77     if (!create.sent) {
 78       mssg = 'No se ha podido enviar el código de validación. Intente de nuevo'
 79       logger.log('error', mssg, {user: req.body.email, operation: 'create', status: 500})
 80       return res.status(500).send(response(false, create.errors, mssg))
 81     } else {
 82       mssg = 'Usuario creado con éxito. El código de validación se ha enviado '
 83       mssg += 'a su correo'
 84       logger.log('info', mssg, {user: req.body.email, operation: 'create', status: 200})
 85       return res.status(200).send(response(true, create.log, mssg))
 86     }
 87   } else if (user.code === 11000) {
 88     mssg = 'El usuario ya existe'
 89     logger.log('error', mssg, {user: req.body.email, operation: 'create', status: 500})
 90     return res.status(500).send(response(false, 'Ya existe usuario', mssg))
 91   } else {
 92     mssg = 'El usuario no ha sido creado debido a un error desconocido'
 93     logger.log('error', mssg, {user: req.body.email, operation: 'create', status: 500})
 94     return res.status(500).send(response(false, 'Error', mssg))
 95   }
 96 }

 98 /**
 99  * Función que verifica la validez de los datos de una solicitud para
100  * crear un nuevo usuario.
101  * @author Francisco Márquez <12-11163@usb.ve>
102  * @private
103  * @param {string} dataRegister
104  * @returns {Verification}
105  */
106 async function verifyCreate (dataRegister) {
107   const validate = validateIn(dataRegister, registerRules, registerMessage)
108   var code = 0
109   var err = ''
110   var message = ''
111   if (dataRegister.email.split('@')[1] !== 'usb.ve') {
112     code = 400
113     err = 'Correo invalido'
114     message = 'Correo debe ser usb.ve'
115     return { code: code, status: false, errors: err, message: message }
116   }
117   const user = await findByEmail(dataRegister.email)
118   if (!(validate.pass && !user)) {
119     if (!validate.pass) {
120       code = 400
121       err = validate.errors
122       message = 'Los datos introducidos no cumplen con el formato requerido'
123     } else if (user.isVerify) {
124       code = 403
125       err = 'Usuario registrado'
126       message = 'Usuario ya se ha registrado exitosamente'
127     } else {
128       code = 409
129       err = 'Usuario debe validarse'
130       message = 'El usuario debe ingresar código enviado a su e-mail'
131       const { sent, errors } = await responseCreate(user, true)
132       if (!sent) {
133         code = 500
134         err = errors
135         message = 'El usuario debía validarse pero ocurrió un error reenviando'
136         message += ' el correo con el código de validación'
137       }
138     }
139     return { code: code, status: false, errors: err, message: message }
140   }
141   return { code: 200, status: true, errors: '', message: '' }
142 }

157 /**
158  * Función que envia el correo de confirmacion para completar el proceso de
159  * registro.
160  * @async
161  * @private
162  * @param {Object} usr
163  * @param {boolean} [already]
164  * @returns {Object}
165  */
166 async function responseCreate (usr, already = false) {
167   const to = usr.email
168   const sb = 'Bienvenido a Pide Cola USB, valida tu cuenta'
169   let code
170   if (already) {
171     code = Math.floor(Math.random() * 90000) + 9999
172     await updateCode(usr.email, code)
173   } else {
174     code = usr.temporalCode
175   }
176   try {
177     return sendEmail(to, sb, template(code, to.split('@')[0])).then(callbackMail)
178   } catch (error) {
179     console.log('ERROR SENDING EMAIL');
180     
181   }
182   
183 }

195   const update = {

222 /**
223  * Endpoint para validar un usuario vía código enviado por e-mail.
224  * No debería modificarse a no ser que se cambie toda lógica detrás del
225  * algoritmo de recomendación.
226  * @function
227  * @async
228  * @public
229  * @param {Object} req - Un HTTP Request
230  * @param {Object} res - Un HTTP Response
231  * @returns {Object}
232  */
233 async function codeValidate (req, res) {
234   var status, err, message
235   if (!(req.body.code && req.body.email)) {
236     err = 'El código es necesario. El email es necesario'
237     message = 'Los datos introducidos no cumplen con el formato requerido'
238     if (!req.body.code && !req.body.email) {
239       status = 400
240     } else if (!req.body.code) {
241       err = 'El código es necesario'
242       status = 403
243     } else {
244       err = 'El email es necesario'
245       status = 401
246     }
247     logger.log('error', message, {user: req.body.email, operation: 'code-validation', status: status})
248     return res.status(status).send(response(false, err, message))
249   }
250   const user = await findByEmail(req.body.email).then(user => {
251     if (!user) return { code: 404, data: 'Usuario no existe' }
252     if (user.isVerify) return { code: 400, data: 'Verificado' }
253     if (user.temporalCode !== parseInt(req.body.code)) {
254       return { code: 401, data: 'Código no coincide' }
255     }
256     user.isVerify = true
257     user.markModified('isVerify')
258     user.save()
259     const token = autentication.generateToken(user.email)
260     return { code: 200, data: [{ tkauth: token }] }
261   }).catch(error => { return { code: 500, data: error } })
262   status = user.code === 200 || user.code === 500 ? user.code : 401
263   if (user.code === 200) {
264     message = 'Usuario verificado'
265   } else if (user.code === 404) {
266     message = 'El usuario debe registrarse primero'
267   } else if (user.code === 400) {
268     message = 'El usuario ya se encuentra verificado'
269   } else if (user.code === 401) {
270     message = 'El código es incorrecto'
271   } else {
272     message = 'Ha ocurrido un error desconocido, por favor intente nuevamente'
273   }
274   logger.log('info', message, {user: req.body.email, operation: 'code-validation', status: 200})
275   return res.status(status).send(response(status === 200, user.data, message))
276 }

282 /**
283  * Endpoint que realiza una consulta sobre la colección Usuario para mostrar
284  * todos los datos asociados a un usuario dado su email.
285  * @public
286  * @param {Object} req - Un HTTP Request
287  * @param {Object} res - Un HTTP Response
288  * @returns {Object}
289  */
290 async function getUserInformation (req, res) {
291   const validate = validateIn(req.secret, emailRules, emailMessage)
292   if (!validate.pass) {
293     const message = 'Los datos no cumplen con el formato requerido'
294     return res.status(401).send(response(false, validate.errors, message))
295   }
296   const usr = await findByEmail(req.secret.email).then(callback)
297   if (!!usr && usr.isVerify) {
298     logger.log('info', 'Perfil encontrado', {user: req.secret.email, operation: 'view-profile', status: 200})
299     return res.status(200).send(response(true, usr, 'Perfil encontrado'))
300   } else {
301     logger.log('error', 'Usuario no existe', {user: req.secret.email, operation: 'view-profile', status: 500})
302     return res.status(500).send(response(false, null, 'Usuario no existe'))
303   }
304 }

310 /**
311  * Endpoint para modificar datos de usuario (nombre, apellido, edad, tlf y carrera)
312  * en la BD.
313  * @public
314  * @param {Object} req - Un HTTP Request
315  * @param {Object} res - Un HTTP Response
316  * @returns {Object}
317  */
318 async function updateUser (req, res) {
319   const validate = validateIn(req, updateRules, updateMessage)
320   var message = 'Debes verificar tu cuenta antes', status = 0, data = null, level = null
321   if (!validate.pass) {
322     level = 'error'
323     message = 'Los datos no cumplen con el formato requerido'
324     logger.log(level, message, {user: req.secret.email, operation: 'update-profile', status: 401})
325     return res.status(401).send(response(false, validate.errors, message))
326   }
327   const query = { $set: req.body }
328   const usr = await updateUserByEmail(req.secret.email, query).then(callback)
329   status = usr && usr.isVerify ? 200 : 500
330   if (usr && usr.isVerify) {
331     data = usr
332     message = 'El Usuario fue actualizado'
333     level = 'info'
334   } else if (!usr.isVerify) {
335     data = 'Usuario no ha sido verificado'
336     level = 'error'
337   } else {
338     message = 'El Usuario no existe'
339     level = 'error'
340   }
341   logger.log(level, message, {user: req.secret.email, operation: 'update-profile', status: status})
342   return res.status(status).send(response(usr && usr.isVerify, data, message))
343 }

360 /**
361  * Endpoint para agregar foto de perfil en el perfil de usuario. Se utiliza
362  * Cloudinary y Multer para el manejo y almacenamiento de imágenes. En la BD
363  * se almacena el URL de Cloudinary donde se encuentra la imagen
364  * @async
365  * @public
366  * @param {Object} req - Un HTTP Request
367  * @param {Object} res - Un HTTP Response
368  * @returns {Object}
369  */
370 async function updateProfilePic (req, res) {
371   const validate = validateIn(req.secret, emailRules, emailMessage)
372   var status = 401, err = '', level = 'error'
373   var message = 'Los datos introducidos no cumplen con el formato requerido'
374   if (!(req.file && validate.pass)) {
375     if (!req.file && !validate.pass) {
376       err = validate.errors.push('La foto de perfil es requerida')
377     } else if (!validate.pass) {
378       err = validate.errors
379     } else if (req.fileValidationError) {
380       err = 'Formatos de imagenes permitidos: jpg, jpeg y png'
381     } else {
382       err = 'La foto de perfil es requerida'
383     }
384     logger.log(level, message, {user: req.secret.email, operation: 'update-profile-pic', status: status})
385     return res.status(status).send(response(false, err, message))
386   }
387   const usr = await findByEmail(req.secret.email)
388     .then(async user => {
389       const picture = await files.uploadFile(req.file.path)
390       if (!picture) return { code: 500, data: 'Error desconocido' }
391       user.$set({ profile_pic: picture.secure_url })
392       const modified = user.save().then((usr, err) => {
393         if (!err) return { code: 200, data: usr }
394         return { code: 501, data: err }
395       })
396       return modified
397     })
398     .catch(error => { return { code: 500, data: error } })
399   status = usr.code === 200 ? 200 : 500
400   if (usr.code === 200) message = 'Foto de perfil agregada', level = 'info'
401   else if (usr.code === 500) message = 'Ocurrió un error en el proceso', level = 'error'
402   else message = 'Foto de perfil no fue agregada', level = 'error'
403   logger.log(level, message, {user: req.secret.email, operation: 'update-profile-pic', status: status})
404   return res.status(status).send(response(status === 200, usr.data, message))
405 }

411 /**
412  * Endpoint para agregar un vehículo al perfil de un usuario.
413  * No debería modificarse a no ser que se cambie toda lógica detrás del
414  * algoritmo de recomendación.
415  * Endpoint para agregar vehiculo en la base de datos. Se actualiza el
416  * respectivo documento de usuario agregando un elemento en el arreglo
417  * vehiculo.
418  * @public
419  * @param {Object} req - Un HTTP Request
420  * @param {Object} res - Un HTTP Response
421  * @returns {Object}
422  */
423 async function addVehicle (req, res) {
424   const validate = validateIn(req, addVehicleRules, addVehicleMessage)
425   var status = 401, err = '', level = 'error'
426   let message = 'Los datos introducidos no cumplen con el formato requerido'
427   if (!(req.file && validate.pass)) {
428     if (!req.file && !validate.pass) {
429       err = validate.errors.push('La foto del vehículo es requerida')
430     } else if (!validate.pass) {
431       err = validate.errors
432     } else if (req.fileValidationError) {
433       err = 'Formatos de imagenes permitidos: jpg, jpeg y png'
434     } else {
435       err = 'La foto del vehículo es requerida'
436     }
437     logger.log(level, message, {user: req.secret.email, operation: 'add-vehicle', status: status})
438     return res.status(status).send(response(false, err, message))
439   }
440   const usr = await findByEmail(req.secret.email)
441     .then(async user => {
442       if (user.vehicles) {
443         if (user.vehicles.find(car => car.plate === req.body.plate)) {
444           return { code: 403, data: user }
445         }
446       }
447       const picture = await files.uploadFile(req.file.path)
448       if (!picture) return { code: 500, data: user }
449       req.body.vehicle_pic = picture.secure_url
450       user.vehicles.push(req.body)
451       user.markModified('vehicles')
452       const modified = user.save().then((usr, err) => {
453         if (!err) return { code: 200, data: usr }
454         return { code: 501, data: err }
455       })
456       return modified
457     })
458     .catch(error => { return { code: 502, data: error } })
459   status = usr.code === 200 || usr.code === 403 ? usr.code : 500
460   if (usr.code === 200) message = 'Vehículo agregado', level = 'info'
461   else if (usr.code === 403) message = 'Vehículo ya existe', level = 'error'
462   else if (usr.code === 500) message = 'Ocurrió un error en el proceso', level = 'error'
463   else message = 'Vehículo no fue agregado', level = 'error'
464   logger.log(level, message, {user: req.secret.email, operation: 'add-vehicle', status: status})
465   return res.status(status).send(response(status === 200, usr.data, message))
466 }

472 /**
473  * Endpoint para eliminar un vehículo asociado a un documento de usuario dado
474  * la placa del vehículo
475  * @public
476  * @param {Object} req - Un HTTP Request
477  * @param {Object} res - Un HTTP Response
478  * @returns {Object}
479  */
480 async function deleteVehicle (req, res) {
481   const validate = validateIn(req, deleteRules, deleteMessage)
482   var status, message, level
483   if (!validate.pass) {
484     message = 'Los datos no cumplen con el formato requerido'
485     logger.log('error', message, {user: req.secret.email, operation: 'delete-vehicle', status: 401})
486     return res.status(401).send(response(false, validate.errors, message))
487   }
488   const usr = await findByEmail(req.secret.email)
489     .then(async user => {
490       if (user.vehicles) {
491         if (!user.vehicles.find(car => car.plate === req.body.plate)) {
492           return { code: 403, data: user }
493         }
494       } else {
495         return { code: 403, data: user }
496       }
497       const query = { $pull: { vehicles: { plate: req.body.plate } } }
498       const modified = user.updateOne(query).then((usr, err) => {
499         if (err) return { code: 500, data: err }
500         return { code: 200, data: usr }
501       })
502       return modified
503     })
504     .catch(error => { return { code: 500, data: error } })
505   status = usr.code === 200 || usr.code === 403 ? usr.code : 500
506   if (usr.code === 200) message = 'Vehículo eliminado', level = 'info'
507   else if (usr.code === 403) message = 'Vehículo no existe', level = 'error'
508   else message = 'Vehículo no existe', level = 'error'
509   logger.log(level, message, {user: req.secret.email, operation: 'delete-vehicle', status: status})
510   return res.status(status).send(response(status === 200, usr.data, message))
511 }
